```
SMOKE-CRM-001  (合约 D-YCZ0331002 + "完蛋，所有人都以为我是绝世高手")
```

```sql
-- 查看合约关联的作品
SELECT DISTINCT cl.pr_channel_id, pc.channel_name, pc.composition_specifications
FROM silverdawn_crm.cp_contract_links cl
JOIN silverdawn_crm.pr_channels pc ON cl.pr_channel_id = pc.id
WHERE cl.contract_id IN (1134801205, 1134801204);

-- 查看交接单记录
SELECT id, channel_id, channel_name, contract_num, delivery_type,
       status, internal_publish, first_publish_time_type, channel_limit
FROM silverdawn_crm.delivery_receipt
WHERE contract_num IN ('D-YCZ0331002', 'DM-YCZ0331001')
  AND del_flag = 0;

-- 查看 AMS 同步结果（提交交接单后验证）

       first_publish_time_type, first_publish_time, channel_limit,
       subtitle_lang_config, dubbing_lang_config

WHERE name IN ('完蛋，所有人都以为我是绝世高手', '第二春', '一剑倾城');
```
